from django.shortcuts import render def calculator_view(request): result = None if request.method == "POST": try: num1 = float(request.POST.get('num1')) num2 = float(request.POST.get('num2')) op = request.POST.get('op') if op == '+': result = num1 + num2 elif op == '-': result = num1 - num2 elif op == '*': result = num1 * num2 elif op == '/': if num2 != 0: result = num1 / num2 else: result = "Error: Division by zero" else: result = "Invalid operation" except (ValueError, TypeError): result = "Invalid input. Please enter valid numbers." return render(request, 'your_template.html', {'result': result})
